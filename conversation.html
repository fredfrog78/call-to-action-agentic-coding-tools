Representative fix (JS) 
// only bind tabs owned by this [data-tabs] group
const tablist = getOwnedTablist(group);
const tabs = Array.from(tablist.querySelectorAll('[role="tab"]'));

// only toggle panels that belong to the same group (supports nested tabs)
const byId = document.getElementById(panelId);
if (byId && byId.closest("[data-tabs]") === group) panel = byId;
